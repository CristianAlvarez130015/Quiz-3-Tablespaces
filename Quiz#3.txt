--1
Create bigfile tablespace Big_tbs_Cristian
DATAFILE 'Punto1' size 1000M
EXTENT MANAGEMENT LOCAL
ONLINE;

--2
CREATE TABLESPACE Quiz
DATAFILE 'Datos1' size 200M,
         'Datos2' size 200M,
         'Datos3' size 200M,
AUTOEXTEND ON NEXT 600M MAXSIZE 1000M
EXTENT MANAGEMENT LOCAL
SEGMENT SPACE MANAGEMENT AUTO
ONLINE;

--3

CREATE PROFILE CAP
LIMIT SESSION UNLIMITED
IDLE_TIME             15
FAILED_LOGIN_ATTEMPTS UNLIMITED
PASSWORD_LIFE_TIME UNLIMITED;


--4
CREATE USER Cristian
IDENTIFIED BY contraseña 
DEFAULT TABLESPACE Quiz
PROFILE CAP;
    
--5
GRANT CONNECT TO cap;
